Class AttendanceSystem
    Attribute:
        attendance[students][days]  // 2D Array
    Method markAttendance(studentId, day, status):
        attendance[studentId][day] = status
    Method updateAttendance(studentId, day, status):
        attendance[studentId][day] = status
    Method generateReport(studentId):
        count = 0
        For each day in attendance[studentId]:
            If status == "P":
                count++
        Print "Student", studentId, "present for", count, "days"

Main():
    system = new AttendanceSystem(5 students, 30 days)
    system.markAttendance(1, 1, "P")
    system.markAttendance(1, 2, "P")
    system.markAttendance(1, 3, "A")

    system.updateAttendance(1, 3, "P")   // correction
    system.generateReport(1)             // total days present
